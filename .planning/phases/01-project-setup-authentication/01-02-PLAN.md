### From 01-RESEARCH.md — Auth Flow Pattern
```javascript
// Lazy auth flow — login triggered on "扭一下" tap
onGashaponTap() {
  if (!this.data.isAuthorized) {
    auth.requestAuth(
      (userInfo) => {
        this.setData({ userInfo, isAuthorized: true });
        this.startGashapon();
      },
      () => {
        // User skipped auth — still allow gashapon
        this.startGashapon();
      }
    );
  } else {
    this.startGashapon();
  }
}

startGashapon() {
  // Proceed to result page
  wx.navigateTo({
    url: `/pages/result/result?brand=${this.data.currentBrand}&milkTea=默认奶茶`
  });
}
```
