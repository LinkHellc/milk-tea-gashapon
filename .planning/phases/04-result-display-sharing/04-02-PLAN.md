<task type="auto">
  <name>Task 3: Add share image generation logic to result.js</name>
  <files>pages/result/result.js</files>
  <action>
Read the existing pages/result/result.js first.

Replace the entire content with:

```javascript
const auth = require('../../utils/auth.js');

Page({
  data: {
    userInfo: null,
    showLoginPrompt: false,
    milkTea: null,
    brand: null,
    description: '经典口味，浓郁奶香',
    capsuleClass: '',
    pearlClass: '',
    nameClass: '',
    cardClass: '',
    pearlGraphicClass: '',
    showPearl: false,
    shareImagePath: ''
  },

  onLoad(options) {
    // Set result data from URL params
    this.setData({
      milkTea: options.milkTea ? decodeURIComponent(options.milkTea) : '默认奶茶',
      brand: options.brandName ? decodeURIComponent(options.brandName) : '未知品牌',
      description: options.description ? decodeURIComponent(options.description) : '经典口味，浓郁奶香'
    });

    // Check auth
    const userInfo = auth.getStoredUserInfo();
    if (userInfo) {
      this.setData({ userInfo, showLoginPrompt: false });
    } else {
      this.setData({ showLoginPrompt: true });
    }

    // Trigger result animation sequence
    this.triggerResultAnimation();
  },

  triggerResultAnimation() {
    // Step 1: Capsule pops in
    setTimeout(() => {
      this.setData({ capsuleClass: 'capsule-popin' });
    }, 200);

    // Step 2: Capsule opens after 0.8s
    setTimeout(() => {
      this.setData({ capsuleClass: 'capsule-open-result', showPearl: true });
      // Pearl bounces in
      this.setData({ pearlClass: 'pearl-bounce' });
      // Name fades in
      this.setData({ nameClass: 'result-name-fadein' });
    }, 1000);

    // Step 3: Result card reveals
    setTimeout(() => {
      this.setData({ cardClass: 'card-reveal' });
    }, 1600);

    // Step 4: Pearl graphic below card appears
    setTimeout(() => {
      this.setData({ pearlGraphicClass: 'pearl-bounce' });
    }, 2000);
  },

  generateShareImage() {
    return new Promise((resolve, reject) => {
      const milkTeaName = this.data.milkTea || '精选奶茶';
      const brandName = this.data.brand || '未知品牌';
      const description = this.data.description || '';

      // Brand colors mapping
      const brandColors = {
        '一点点': '#71C685',
        '喜茶': '#F5A6A0',
        '霸王茶姬': '#B89B85',
        '古茗': '#5B9A6E',
        '茶百道': '#5B9ECF',
        'CoCo都可': '#E8A87C',
        'KOI': '#D4A84B',
        '瑞幸': '#5B8ABE',
        '沪上阿姨': '#B87CBB'
      };
      const brandColor = brandColors[brandName] || '#F5A6A0';

      // Get canvas context
      const query = wx.createSelectorQuery();
      query.select('#shareCanvas')
        .fields({ node: true, size: true })
        .exec((res) => {
          if (!res[0] || !res[0].node) {
            reject(new Error('Canvas not found'));
            return;
          }

          const canvas = res[0].node;
          const ctx = canvas.getContext('2d');
          const dpr = wx.getSystemInfoSync().pixelRatio;

          // Set canvas size (600rpx x 480rpx, convert to physical pixels)
          const width = 300;  // 600rpx / 2 = 300px at base
          const height = 240;  // 480rpx / 2 = 240px
          canvas.width = width * dpr;
          canvas.height = height * dpr;
          ctx.scale(dpr, dpr);

          // Draw background gradient
          const gradient = ctx.createLinearGradient(0, 0, 0, height);
          gradient.addColorStop(0, '#fff5f5');
          gradient.addColorStop(1, brandColor + '33');
          ctx.fillStyle = gradient;
          ctx.fillRect(0, 0, width, height);

          // Draw rounded rectangle background
          ctx.fillStyle = '#ffffff';
          ctx.shadowColor = 'rgba(0,0,0,0.1)';
          ctx.shadowBlur = 10;
          ctx.shadowOffsetY = 4;
          this.roundRect(ctx, 30, 30, width - 60, height - 80, 16);
          ctx.fill();
          ctx.shadowBlur = 0;
          ctx.shadowOffsetY = 0;

          // Draw pearl (circle with gradient)
          const pearlGradient = ctx.createRadialGradient(150, 70, 0, 150, 70, 30);
          pearlGradient.addColorStop(0, '#ffffff');
          pearlGradient.addColorStop(0.3, '#F5A6A0');
          pearlGradient.addColorStop(1, '#E88888');
          ctx.beginPath();
          ctx.arc(150, 70, 25, 0, 2 * Math.PI);
          ctx.fillStyle = pearlGradient;
          ctx.fill();

          // Draw pearl highlight
          ctx.beginPath();
          ctx.arc(142, 62, 8, 0, 2 * Math.PI);
          ctx.fillStyle = 'rgba(255,255,255,0.6)';
          ctx.fill();

          // Draw milk tea name
          ctx.fillStyle = '#333333';
          ctx.font = 'bold 24px sans-serif';
          ctx.textAlign = 'center';
          ctx.fillText(milkTeaName, width / 2, 130);

          // Draw brand name
          ctx.fillStyle = brandColor;
          ctx.font = '16px sans-serif';
          ctx.fillText(brandName, width / 2, 155);

          // Draw description
          ctx.fillStyle = '#999999';
          ctx.font = '12px sans-serif';
          ctx.fillText(description, width / 2, 180);

          // Draw bottom watermark
          ctx.fillStyle = '#cccccc';
          ctx.font = '10px sans-serif';
          ctx.fillText('via 今天喝什么', width / 2, 200);

          // Convert to temp file path
          wx.canvasToTempFilePath({
            canvas: canvas,
            success: (res) => {
              this.setData({ shareImagePath: res.tempFilePath });
              resolve(res.tempFilePath);
            },
            fail: (err) => {
              console.error('Canvas to temp file failed:', err);
              reject(err);
            }
          }, this);
        });
    });
  },

  roundRect(ctx, x, y, w, h, r) {
    ctx.beginPath();
    ctx.moveTo(x + r, y);
    ctx.lineTo(x + w - r, y);
    ctx.quadraticCurveTo(x + w, y, x + w, y + r);
    ctx.lineTo(x + w, y + h - r);
    ctx.quadraticCurveTo(x + w, y + h, x + w - r, y + h);
    ctx.lineTo(x + r, y + h);
    ctx.quadraticCurveTo(x, y + h, x, y + h - r);
    ctx.lineTo(x, y + r);
    ctx.quadraticCurveTo(x, y, x + r, y);
    ctx.closePath();
  },

  onLoginTap() {
    auth.requestAuth(
      (userInfo) => {
        this.setData({ userInfo, showLoginPrompt: false });
        wx.showToast({ title: '登录成功', icon: 'success' });
      },
      () => {
        wx.showToast({ title: '登录已取消', icon: 'none' });
      }
    );
  },

  onRetryTap() {
    wx.navigateBack({ delta: 1 });
  },

  onShareTap() {
    // Generate share image then trigger WeChat share menu
    wx.showShareMenu({
      withShareTicket: true
    });
    // Note: onShareAppMessage will be called when user actually shares
  },

  onShareAppMessage(res) {
    const milkTeaName = this.data.milkTea || '精选奶茶';
    const brandName = this.data.brand || '';
    const description = this.data.description || '';

    // Generate share image synchronously for share card
    // The image will be attached if available
    return {
      title: `我抽到了${brandName}的${milkTeaName}！`,
      desc: description,
      path: `/pages/result/result?brandId=&milkTea=${encodeURIComponent(milkTeaName)}&brandName=${encodeURIComponent(brandName)}&description=${encodeURIComponent(description)}`,
      imageUrl: this.data.shareImagePath || ''
    };
  }
});
```
